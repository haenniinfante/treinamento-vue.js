new Vue({
 el:'#app',
 data:{
    novoNome: '',
    nomes: [],
 },
 methods:{
   adicionarNome: function(){
       this.nomes.push(this.novoNome);      
      // this.novoNome='';
   }
 },
 filters:{
   capitalizar: function(value){
    return value.toUppercase();
   }
 }
 })
<script src="https://cdn.jsdelivr.net/npm/vue/dist/vue.js"></script>
<div id ='app'>
  <input type="text" v-model="novoNome">
  <button @click="adicionarNome" >
    Adicionar Nome +
  </button>
<ul>
 <li v-for="nome in nomes">{{ nome }}</li>
</ul>
</div>
