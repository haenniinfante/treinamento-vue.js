new Vue({
 el:'#app',
 data:{
    numCliques:0,
 },
 methods:{
   Incrementar: function(){
       this.numCliques++;
   }
 },
 computed: {
   contador(){
    return this.numCliques*10;
   }
 }
 })
<script src="https://cdn.jsdelivr.net/npm/vue/dist/vue.js"></script>
<div id ='app'>
  <p>Contador:{{contador}}</p>
  <p>Cliques:{{numCliques}}</p>
  <button v-on:click="Incrementar" >
    Incrementar
  </button>
</div>
