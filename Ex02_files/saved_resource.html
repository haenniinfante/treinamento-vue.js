new Vue({
 el:'#app',
 data:{
    titulo: 'Minha aplicação em Vue',
    paragrafo:'Das hat Karl Marx gesagt',
 },
 methods: {
 
   alterarTitulo:function () {
     this.titulo = 'Oba , estou aprendendo!'
   },
 }
    
})

<div id ='app'>
  <h3>{{ titulo }}</h3>
  <p>
      {{ paragrafo }}
  </p>
